RESTful API sample to insert data into table

source code : 

    var send_data = [];

    for(var i=1; i <= 100; i++)
    {
        send_data.push([i, 'abcd-'+i]);
    }

    $.ajax({
        type:'post',
        async:true,
        url: 'http://127.0.0.1:5657/machbase', //set MWA server ip and port
        contentType: 'application/json; charset=UTF-8',
        dataType: 'json',
        data: JSON.stringify({"name":"rest_sample","values":send_data}),
        beforeSend: function() {
        },
        success: function(resp) {
            $('#result').text(JSON.stringify(resp));
        },
        error: function(request,status,err) {
            $('#result').text(request.responseText);
        },
        complete: function() {
        }
    });
