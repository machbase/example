RESTful API sample to create table

source code : 

    var sql = "create table rest_sample(c1 int, c2 varchar(20))";

    $.ajax({
        type:'get',
        async:true,
        url: 'http://127.0.0.1:5657/machbase', //set MWA server ip and port
        data: 'q='+encodeURIComponent(sql),
        beforeSend: function() {
        },
        success: function(resp) {
            $('#result').text(JSON.stringify(resp));
        },
        error: function(request,status,err) {
            $('#result').text(request.responseText);
        },
        complete: function() {
        }
    });
